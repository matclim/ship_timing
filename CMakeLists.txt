cmake_minimum_required(VERSION 3.20)
project(phlex_time_slicing LANGUAGES CXX)

# Phlex transitive deps -- imported by name in phlexConfig.
find_package(Boost  REQUIRED COMPONENTS json)
find_package(TBB    REQUIRED)
find_package(spdlog REQUIRED)

find_package(phlex  REQUIRED)
find_package(ROOT   REQUIRED COMPONENTS Core RIO Tree)

# include(${ROOT_USE_FILE}) overrides our C++ standard with whatever ROOT
# was built for (C++17 here), which breaks Phlex's C++20+ headers. Don't
# use it. We keep ROOT_GENERATE_DICTIONARY available implicitly through
# find_package(ROOT) which loads RootMacros.cmake.

# Set the C++ standard AFTER everything else so nothing overrides it.
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ---------------------------------------------------------------------------
# ROOT dictionary for our data product types.
#
# Phlex's form_module looks up TDictionary::GetDictionary(type) and refuses
# to write any type that has no dictionary. So we have to generate one with
# rootcling for every C++ type that flows through Phlex products.
#
# Output: a generated source file (and a _rdict.pcm + .rootmap installed
# beside it). We compile the .cxx into the plugins so the dictionary
# registers itself at plugin-load time.
# ---------------------------------------------------------------------------
ROOT_GENERATE_DICTIONARY(
    G__time_slicing_dict       # base name of generated files
    time_slicing.hpp           # headers to read for the dictionary
    LINKDEF LinkDef.h
)

# ---------------------------------------------------------------------------
# Build the dictionary as a real SHARED library named "time_slicing_dict".
# ROOT's auto-load mechanism (TInterpreter::AutoLoad) reads the .rootmap
# file alongside it and maps each registered type to this lib's filename,
# so this MUST be a real .so on disk (not an OBJECT library), with the
# exact name matching the .rootmap entries.
#
# Both plugins also link this so the dictionary's static-init runs at
# plugin load and the types are immediately available, without needing
# auto-load.
# ---------------------------------------------------------------------------
add_library(time_slicing_dict SHARED
    G__time_slicing_dict.cxx
    time_slicing.cpp           # the .cxx generated by rootcling references
                                # the actual class implementations
)
target_include_directories(time_slicing_dict PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(time_slicing_dict PRIVATE
    ROOT::Core ROOT::RIO ROOT::Tree
)
set_target_properties(time_slicing_dict PROPERTIES
    CXX_STANDARD 23
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
    POSITION_INDEPENDENT_CODE ON
    PREFIX "lib"           # standard prefix; rootmap expects libtime_slicing_dict.so
)

# Plugin: muon_hits_source
add_library(muon_hits_source MODULE
    muon_hits_source.cpp
)
target_include_directories(muon_hits_source PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(muon_hits_source PRIVATE
    phlex::module
    time_slicing_dict
    ROOT::Core ROOT::RIO ROOT::Tree
)
set_target_properties(muon_hits_source PROPERTIES
    CXX_STANDARD 23
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
)

# Plugin: time_module
add_library(time_module MODULE
    time_module.cpp
)
target_include_directories(time_module PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(time_module PRIVATE
    phlex::module
    time_slicing_dict
    ROOT::Core ROOT::RIO ROOT::Tree
)
set_target_properties(time_module PROPERTIES
    CXX_STANDARD 23
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
)
